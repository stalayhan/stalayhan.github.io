How to integrate Catch2 ?

You can directly download the single header file(v2.13.3) using this link. https://github.com/catchorg/Catch2/releases/download/v2.13.3/catch.hpp

If you want to discover the source code please checkout the library with the v2.x branch and copy the catch.hpp file on your source folder.

git clone -b v2.x https://github.com/catchorg/Catch2.git
ls -all Catch2/single_include/catch2/catch.hpp

Here is my test folder. Please find the source code below.

.
├── build
├── CMakeLists.txt
├── inc
│   └── catch.hpp
└── src
    └── main.cpp

    3 directories, 2 files

CMakeLists.txt

cmake_minimum_required(VERSION 3.6)
project(catchtest)

include_directories(${library_INCLUDE_DIRS})

add_executable(catchtest src/main.cpp)

target_link_libraries(catchtest ${library_LIBRARIES})

main.cpp

#define CATCH_CONFIG_MAIN
#include "../inc/catch.hpp"

unsigned int Factorial( unsigned int number ) {
    return number <= 1 ? number : Factorial(number-1)*number;
}

TEST_CASE( "Factorials are computed", "[factorial]" ) {
    REQUIRE( Factorial(1) == 1 );
    REQUIRE( Factorial(2) == 2 );
    REQUIRE( Factorial(3) == 6 );
    REQUIRE( Factorial(10) == 3628800 );
}
